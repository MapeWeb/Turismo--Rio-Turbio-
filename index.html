/* --- FUNCIÓN DE NAVEGACIÓN CORREGIDA --- */ let origen = 'audioguias'; const playerContainer = document.getElementById('player-container'); const globalPlayer = document.getElementById('global-player'); function showPage(id) { document.querySelectorAll('.page').forEach(p => p.style.display = 'none'); document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active')); const target = document.getElementById(id); if(target) { target.style.display = 'block'; if(id !== 'modulo-detalle') { origen = id; const btn = document.getElementById('btn-' + id); if(btn) btn.classList.add('active'); } } window.scrollTo(0,0); } function verRestaurante(titulo, imgP, galeria, ubi, resena, pdf) { document.getElementById('det-titulo').innerText = titulo; document.getElementById('det-img-principal').src = imgP; document.getElementById('det-ubi').innerText = ubi; document.getElementById('det-resena').innerText = resena; const galContainer = document.getElementById('det-galeria'); galContainer.innerHTML = ""; if(galeria && galeria.length > 0) { galeria.forEach(src => { const img = document.createElement('img'); img.src = src; img.onclick = () => document.getElementById('det-img-principal').src = src; galContainer.appendChild(img); }); } const acciones = document.getElementById('acciones-gastronomia'); if(pdf && pdf !== "") { acciones.style.display = 'flex'; document.getElementById('link-ver').href = pdf; document.getElementById('link-descargar').href = pdf; } else { acciones.style.display = 'none'; } showPage('modulo-detalle'); } function volver() { showPage(origen); } function playAudio(url, title) { if(!url) { alert("Audio pronto disponible"); return; } playerContainer.classList.add('visible'); document.body.style.paddingBottom = '160px'; globalPlayer.src = url; document.getElementById('playing-title').innerText = "Reproduciendo: " + title; globalPlayer.play(); } globalPlayer.onended = function() { playerContainer.classList.remove('visible'); document.body.style.paddingBottom = '20px'; globalPlayer.pause(); globalPlayer.src = ""; };
